namespace Siad007\VersionControl\HG\Command;

/**
 * Base command.
 */
class BaseCommand extends AbstractCommand
{
    /**
     * Available arguments for this command.
     *
     * @var array $arguments
     */
    protected $arguments = array();

    /** @var mixed $options */
    protected $options = array();

    /**
     * {@inheritdoc}
     */
    public function __toString()
    {
        return $this->name;
    }
}
